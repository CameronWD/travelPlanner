# Chapters Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add optional, named, coloured date-range "Chapters" that group a long Trip into legs, with per-chapter roll-ups in the Itinerary, Budget and Summary, and a chapter label in the Today view.

**Architecture:** A `Chapter` is a thin row (`tripId, name, colour, startDate, endDate, sortOrder`). Membership is **computed from dates at read time** — no foreign keys on Stop/Transport/Cost (see `docs/adr/0008-chapters-as-computed-date-range-bands.md`). A Stop belongs to the Chapter whose `[startDate, endDate]` covers its **arrive date**; a Transport whose two ends fall in different chapters (or to/from home, i.e. a null stop) is **between-legs travel** — shown on the seam and as its own Budget line. Chapters are optional, cannot overlap, and need not cover the trip; uncovered dates are **Ungrouped**.

**Tech Stack:** Next.js (App Router, RSC + client components), Prisma 7 + Postgres (driver adapter), Zod validation, Vitest (Prisma mocked), Tailwind, Leaflet (`route-map`), `motion/react`.

**Key conventions to mirror (verified against the codebase):**
- Dates are `"YYYY-MM-DD"` strings; string comparison is chronological. Money is `Int` minor units. No Prisma `enum`/`Json` — values are `String` validated by Zod (`lib/enums.ts`, `lib/categories.ts`).
- Colour palette pattern: `lib/categories.ts` (data) + class map in the chip component (`components/trip/category-pill.tsx`).
- Server actions: `"use server"`; `requireTripAccess(tripId)` first; `schema.safeParse`; return `{ success: true } | { success: false; errors }`; `revalidatePath`. Reorders use `$transaction` + `FOR UPDATE` (ADR 0007) — **not needed here** (chapters order by `startDate`).
- Tests: pure functions tested directly; DB-touching code mocks `@/lib/db` via `vi.hoisted` + `vi.mock` (pattern in `server/actions/stops.test.ts`). The test sandbox has no Postgres.
- Date helpers in `lib/dates.ts`: `isDateWithin(date, start, end)` (inclusive), `daysBetween(a, b)`, `addDays`, `formatLongDate`, `nightsBetween`. Budget core is the pure `buildBudget()` in `lib/budget.ts` with helper `stopIdForCost()`.

  return chapterColourMeta(value).swatch;

/** First palette colour not already used; cycles to the first when all are used. */
export function nextChapterColour(used: readonly string[]): ChapterColour {
  const usedSet = new Set(used);
  const free = CHAPTER_COLOUR_VALUES.find((v) => !usedSet.has(v));
  return free ?? CHAPTER_COLOUR_VALUES[0];
}

import { isDateWithin } from "./dates";

export interface ChapterLike {
  id: string;
  name: string;
  colour: string;
  startDate: string; // YYYY-MM-DD
  endDate: string;   // YYYY-MM-DD
}

export interface StopLike {
  id: string;
  arriveDate: string;
  departDate: string;
  country?: string | null;
  sortOrder: number;
}

export interface TransportLike {
  id: string;
  fromStopId?: string | null;
  toStopId?: string | null;
  depAt?: Date | string | null;
}

/** Chapters sorted by start date (ascending). */
function sortedByStart<T extends { startDate: string }>(chapters: readonly T[]): T[] {
  return [...chapters].sort((a, b) => a.startDate.localeCompare(b.startDate));
}

/** The chapter whose [startDate, endDate] covers `dateISO`, or null. Earliest-starting wins a shared boundary day. */
export function chapterForDate<T extends ChapterLike>(dateISO: string, chapters: readonly T[]): T | null {
  for (const c of sortedByStart(chapters)) {
    if (isDateWithin(dateISO, c.startDate, c.endDate)) return c;
  }
  return null;
}

/** A stop is placed by its arrive date (the band it starts in). */
export function chapterForStop<T extends ChapterLike>(stop: StopLike, chapters: readonly T[]): T | null {
  return chapterForDate(stop.arriveDate, chapters);
}

/** Inclusive date-range overlap (used to forbid overlapping chapters). */
export function chaptersOverlap(a: { startDate: string; endDate: string }, b: { startDate: string; endDate: string }): boolean {
  return a.startDate <= b.endDate && b.startDate <= a.endDate;
}

export interface ChapterGroup<C extends ChapterLike, S extends StopLike> {
  chapter: C | null; // null = Ungrouped run
  stops: S[];
}

/** Walk stops in given order, starting a new group whenever the covering chapter id changes. */
export function groupStopsByChapter<C extends ChapterLike, S extends StopLike>(stops: readonly S[], chapters: readonly C[]): ChapterGroup<C, S>[] {
  const groups: ChapterGroup<C, S>[] = [];
  for (const stop of stops) {
    const chapter = chapterForStop(stop, chapters);
    const last = groups[groups.length - 1];
    const lastId = last ? (last.chapter?.id ?? null) : undefined;
    if (last && lastId === (chapter?.id ?? null)) {
      last.stops.push(stop);
    } else {
      groups.push({ chapter, stops: [stop] });
    }
  }
  return groups;
}

/** The chapter id a transport rolls up under, or null if it is between-legs. */
export function chapterIdForTransport(
  transport: TransportLike,
  chapters: readonly ChapterLike[],
  stopsById: Record<string, StopLike>,
): string | null {
  const from = transport.fromStopId ? stopsById[transport.fromStopId] : undefined;
  const to = transport.toStopId ? stopsById[transport.toStopId] : undefined;
  if (!from || !to) return null; // a home/missing endpoint -> between-legs
  const fromCh = chapterForStop(from, chapters);
  const toCh = chapterForStop(to, chapters);
  if (fromCh && toCh && fromCh.id === toCh.id) return fromCh.id;
  return null;
}

export function isTransportBetweenLegs(
  transport: TransportLike,
  chapters: readonly ChapterLike[],
  stopsById: Record<string, StopLike>,
): boolean {
  return chapterIdForTransport(transport, chapters, stopsById) === null;
}

export interface ChapterRun {
  name: string;
  startDate: string;
  endDate: string;
}

/** Propose one run per consecutive same-country block (country-less stops break/skip a run). */
export function suggestChapterRuns(stops: readonly StopLike[]): ChapterRun[] {
  const ordered = [...stops].sort((a, b) => a.arriveDate.localeCompare(b.arriveDate));
  const runs: ChapterRun[] = [];
  let current: ChapterRun | null = null;
  let currentCountry: string | null = null;
  for (const stop of ordered) {
    const country = stop.country?.trim() || null;
    if (country && country === currentCountry && current) {
      current.endDate = stop.departDate;
    } else if (country) {
      current = { name: country, startDate: stop.arriveDate, endDate: stop.departDate };
      currentCountry = country;
      runs.push(current);
    } else {
      current = null;
      currentCountry = null;
    }
  }
  return runs;
}

```ts
"use server";

import { notFound } from "next/navigation";
import { revalidatePath } from "next/cache";
import { db } from "@/lib/db";
import { requireTripAccess } from "@/lib/guards";
import { chapterSchema, type ChapterInput } from "@/lib/validations/chapter";
import { chaptersOverlap, suggestChapterRuns } from "@/lib/chapters";
import { nextChapterColour } from "@/lib/chapter-colours";

export type ChapterActionResult =
  | { success: true }
  | { success: false; errors: Record<string, string[]> };

function validationErrors(error: { flatten(): { fieldErrors: Record<string, string[] | undefined> } }): ChapterActionResult {
  const fieldErrors: Record<string, string[]> = {};
  for (const [k, msgs] of Object.entries(error.flatten().fieldErrors)) fieldErrors[k] = msgs ?? [];
  return { success: false, errors: fieldErrors };
}

function revalidateChapterPaths(tripId: string) {
  for (const p of ["", "/budget", "/summary", "/today", "/settings", "/calendar"]) {
    revalidatePath(`/trips/${tripId}${p}`);
  }
}

async function requireChapterAccess(chapterId: string): Promise<{ id: string; tripId: string }> {
  const chapter = await db.chapter.findUnique({ where: { id: chapterId }, select: { id: true, tripId: true } });
  if (!chapter) notFound();
  await requireTripAccess(chapter.tripId);
  return chapter;
}

async function firstOverlap(tripId: string, range: { startDate: string; endDate: string }, excludeId?: string) {
  const siblings = await db.chapter.findMany({ where: { tripId }, select: { id: true, startDate: true, endDate: true } });
  return siblings.find((s) => s.id !== excludeId && chaptersOverlap(s, range)) ?? null;
}

async function nextSortOrder(tripId: string): Promise<number> {
  const all = await db.chapter.findMany({ where: { tripId }, select: { startDate: true } });
  return all.length; // order is by startDate at read time; sortOrder is a stable tiebreak only
}

export async function createChapter(tripId: string, input: ChapterInput): Promise<ChapterActionResult> {
  await requireTripAccess(tripId);
  const parsed = chapterSchema.safeParse(input);
  if (!parsed.success) return validationErrors(parsed.error);

  if (await firstOverlap(tripId, parsed.data)) {
    return { success: false, errors: { startDate: ["Chapters cannot overlap another chapter's dates"] } };
  }
  await db.chapter.create({ data: { tripId, ...parsed.data, sortOrder: await nextSortOrder(tripId) } });
  revalidateChapterPaths(tripId);
  return { success: true };
}

export async function updateChapter(chapterId: string, input: ChapterInput): Promise<ChapterActionResult> {
  const chapter = await requireChapterAccess(chapterId);
  const parsed = chapterSchema.safeParse(input);
  if (!parsed.success) return validationErrors(parsed.error);

  if (await firstOverlap(chapter.tripId, parsed.data, chapterId)) {
    return { success: false, errors: { startDate: ["Chapters cannot overlap another chapter's dates"] } };
  }
  await db.chapter.update({ where: { id: chapterId }, data: parsed.data });
  revalidateChapterPaths(chapter.tripId);
  return { success: true };
}

export async function deleteChapter(chapterId: string): Promise<ChapterActionResult> {
  const chapter = await requireChapterAccess(chapterId);
  await db.chapter.delete({ where: { id: chapterId } });
  revalidateChapterPaths(chapter.tripId);
  return { success: true };
}

export async function suggestChaptersFromCountries(tripId: string): Promise<ChapterActionResult> {
  await requireTripAccess(tripId);
  const [stops, existing] = await Promise.all([
    db.stop.findMany({ where: { tripId }, select: { id: true, arriveDate: true, departDate: true, country: true, sortOrder: true } }),
    db.chapter.findMany({ where: { tripId }, select: { id: true, colour: true, startDate: true, endDate: true } }),
  ]);

  const runs = suggestChapterRuns(stops);
  const usedColours = existing.map((c) => c.colour);
  const data: { tripId: string; name: string; colour: string; startDate: string; endDate: string; sortOrder: number }[] = [];

  for (const run of runs) {
    const overlapsExisting = existing.some((c) => chaptersOverlap(c, run)) || data.some((d) => chaptersOverlap(d, run));
    if (overlapsExisting) continue; // additive: only fill Ungrouped spans
    const colour = nextChapterColour([...usedColours, ...data.map((d) => d.colour)]);
    data.push({ tripId, name: run.name, colour, startDate: run.startDate, endDate: run.endDate, sortOrder: existing.length + data.length });
  }

  if (data.length > 0) await db.chapter.createMany({ data });
  revalidateChapterPaths(tripId);
  return { success: true };
}
```
